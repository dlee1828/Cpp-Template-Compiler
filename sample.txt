function get_next_collatz(n) {
    n_mod_2 = n % 2;
    if (n_mod_2 == 0) {
        return n / 2;
    }
    else {
        n = n * 3
        n = n + 1
        return n
    }
}

function print_collatz_sequence(n) {
    while (n != 1) {
        print(n)
        n = get_next_collatz(n)
    }
    print(1)
}

print_collatz_sequence(100)
